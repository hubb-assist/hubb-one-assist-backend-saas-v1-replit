def delete(self, patient_id: UUID, subscriber_id: UUID) -> bool:
    patient = self.db.query(Patient).filter_by(id=patient_id, subscriber_id=subscriber_id, is_active=True).first()
    if not patient:
        return False
    patient.is_active = False
    self.db.commit()
    return True
